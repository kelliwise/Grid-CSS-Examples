.wrapper4 .header {
        grid-area: header;
    }
.wrapper4 .sidebar {
        grid-area: sidebar;
    }
.wrapper4 .cta {
        grid-area: cta;
    }
.wrapper4 .navigation {
        grid-column: 1/4;
        grid-row: 2;
    }
.wrapper4 .content { 
        grid-area: content;
    }
.wrapper4 .footer {
        grid-area: footer;
    }

.wrapper4 {
  display: grid;
  grid-template-columns: 33% 33% 33%;
  grid-gap: 10px;
  background-color: #fff;
  color: #444;
  grid-template-areas:
  "header header header"
  "navigation navigation navigation"
  "content content sidebar"
  "cta ...... sidebar"
  "footer footer footer";
}
